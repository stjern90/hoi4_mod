		name = soviet.2.a

			limit = {
				561 = { 
					is_owned_by = SOV 
					any_neighbor_state = { is_owned_by = JAP } 
				}
			}
			561 = { add_manpower = -30000 }
		}		
		if = {
			limit = { 
				NOT = {

				limit = { any_neighbor_state = { is_owned_by = JAP } }

			}

	}
}
