country_event = {
    id =lend_lease.1
    title = lend_lease.1.t
    desc = lend_lease.1.d
    picture = GFX_report_event_japanese_soldiers_02

    fire_only_once = no
 
    trigger = {
        AND = {
            is_ai = yes
			NOT = { has_idea = Arms_Shipment_Sent }
			NOT = { has_idea = Arms_Shipment_New_Deal }
            has_government = fascism
            has_equipment = {
            infantry_equipment > 500
            }
        }
		JAP = { has_equipment = {
                    infantry_equipment < 200
                }
		}
		
		JAP = { NOT = { has_idea = Arms_Shipment } }
     }
    mean_time_to_happen = {
        days = 2
    }
 
    immediate = {
            random_country = {
                limit = {
                has_government = fascism
               
                NOT = { has_war_with = JAP } 
                }
			random_list = {
				80 = { 
				 send_equipment = {
					target = JAP			
					type = infantry_equipment
					amount = 500
                }     
				}
				20 = { 
				 send_equipment = {
					target = JAP			
					type = infantry_equipment
					amount = 1000
                }     
				}
              }
    }
	}
        option = {
        name = lend_lease.1.a
		add_timed_idea = {
				idea = Arms_Shipment_Sent
				days = 90
			}
		JAP = {
			country_event = { days = 2 id = lend_lease.101 }
		}
        }
		 option = {
        name = lend_lease.1.b
		add_timed_idea = {
				idea = Arms_Shipment_New_Deal
				days = 45
			}
		
        }
}
