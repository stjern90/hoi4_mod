country_event = {
	id = politicalpower.1
	title = politicalpower.1.t
	desc = politicalpower.1.u # event to choose what to improve
	
	picture = GFX_report_event_german_speech

	trigger = {
		date > 1936.1.1
		has_political_power > 290 #typically an average 6 month supply
		NOT = { has_country_flag = political_power_update }
		NOT = { has_country_flag = political_power_disabled }
	}
	
	mean_time_to_happen = {
        days = 2
           modifier = {
			factor = 0.5
			date > 1940.1.1 
		}
	}
	
	immediate = { 
		hidden_effect = { set_country_flag = political_power_update }
	}

	option = {
		name = politicalpower.1.a # improve research (research speed)
		
		trigger = {
			NOT = { has_idea = politicalpower_research10 }
		}
		
		ai_chance = {
			factor = 40
		}
		
		set_country_flag = political_power_research
		add_political_power = -200
	}
	option = {
		name = politicalpower.1.b # improve industrial production (mil/civ factories or production output ideas)
		
		ai_chance = {
			factor = 10
		}
		
		set_country_flag = political_power_production
		add_political_power = -200
	}
	option = {
		name = politicalpower.1.c # improve resource production (resource extraction idea, or mine specific materials )
		
		ai_chance = {
			factor = 5
		}
		
		set_country_flag = political_power_resource
		add_political_power = -200
	}
	option = {
		name = politicalpower.1.d # Inspire the People (add national unity or manpower idea)
		
		ai_chance = {
			factor = 10
		}
		
		set_country_flag = political_power_morale
		add_political_power = -200
	}
	option = {
		name = politicalpower.1.e # Improve Army (add army strengthening idea)
		
		trigger = {
			NOT = { has_idea = politicalpower_army10 }
		}
		
		ai_chance = {
			factor = 30
		}
		
		set_country_flag = political_power_army
		add_political_power = -200
	}
	option = {					
		name = politicalpower.1.f # Improve Navy (navy strengthening idea)
		
		trigger = {
			NOT = { has_idea = politicalpower_navy10 }
		}
