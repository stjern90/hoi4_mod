				NOT = { alliance_strength_ratio > 0.75 }
			}

			AND = {
				alliance_naval_strength_ratio > 1.5
				NOT = { alliance_strength_ratio > 1 }
			}

			AND = {
				alliance_naval_strength_ratio > 1
				NOT = { alliance_strength_ratio > 1.25 }
			}

			NOT = { alliance_naval_strength_ratio > 1 }
